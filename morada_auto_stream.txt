    "import json\n",
    "import random\n",
    "import threading\n",
    "from datetime import datetime\n",
    "from kafka import KafkaProducer\n",
    "\n",
    "def run_producer():\n",
    "    producer = KafkaProducer(\n",
    "        bootstrap_servers=['localhost:9092'],\n",
    "        value_serializer=lambda v: json.dumps(v).encode('utf-8')\n",
    "    )\n",
    "    nomes_br = [\"Ana Souza\", \"Bruno Lima\", \"Carlos Silva\", \"Daniela Costa\", \"Eduardo Pereira\"]\n",
    "    pagamentos = [\"pix\", \"credit_card\", \"split_bill\"]\n",
    "    \n",
    "    while True:\n",
    "        booking_event = {\n",
    "            \"id\": str(random.randint(100000, 999999)),\n",
    "            \"listing_id\": str(random.randint(1, 50)),\n",
    "            \"primary_guest_name\": random.choice(nomes_br),\n",
    "            \"primary_guest_cpf\": f\"{random.randint(100,999)}.{random.randint(100,999)}.{random.randint(100,999)}-{random.randint(10,99)}\",\n",
    "            \"primary_guest_phone\": f\"+55 (11) 9{random.randint(8000,9999)}-{random.randint(1000,9999)}\",\n",
    "            \"total_price\": float(round(random.uniform(300.0, 2000.0), 2)),\n",
    "            \"payment_type\": random.choice(pagamentos),\n",
    "            \"timestamp\": datetime.now().strftime(\"%Y-%m-%d %H:%M:%S\")\n",
    "        }\n",
    "        producer.send('morada-bookings', booking_event)\n",
    "        time.sleep(random.uniform(2.0, 4.0))\n",
    "\n",
    "prod_thread = threading.Thread(target=run_producer, daemon=True)\n",
    "prod_thread.start()\n",
    "print(\"Simulador de reservas (Produtor Kafka) rodando em background enviando eventos a cada 2-4s...\")"
   ]

    "from pyspark.sql import SparkSession\n",
    "from pyspark.sql.functions import col, from_json, regexp_replace, when, expr\n",
    "from pyspark.sql.types import StructType, StructField, StringType, DoubleType, IntegerType\n",
    "\n",
    "# Inicializar a sessão Spark com suporte ao Kafka\n",
    "spark = SparkSession.builder \\\n",
    "    .appName(\"MoradaStreamingNotebook\") \\\n",
    "    .config(\"spark.jars.packages\", \"org.apache.spark:spark-sql-kafka-0-10_2.12:3.5.0\") \\\n",
    "    .config(\"spark.sql.shuffle.partitions\", \"2\") \\\n",
    "    .getOrCreate()\n",
    "    \n",
    "booking_schema = StructType([\n",
    "    StructField(\"id\", StringType(), True),\n",
    "    StructField(\"listing_id\", StringType(), True),\n",
    "    StructField(\"primary_guest_name\", StringType(), True),\n",
    "    StructField(\"primary_guest_cpf\", StringType(), True),\n",
    "    StructField(\"primary_guest_phone\", StringType(), True),\n",
    "    StructField(\"total_price\", DoubleType(), True),\n",
    "    StructField(\"payment_type\", StringType(), True),\n",
    "    StructField(\"timestamp\", StringType(), True)\n",
    "])\n",
    "\n",
    "# Leitura streaming do Kafka\n",
    "df_stream = spark.readStream \\\n",
    "    .format(\"kafka\") \\\n",
    "    .option(\"kafka.bootstrap.servers\", \"localhost:9092\") \\\n",
    "    .option(\"subscribe\", \"morada-bookings\") \\\n",
    "    .option(\"startingOffsets\", \"latest\") \\\n",
    "    .load()\n",
    "\n",
    "# Deserialização\n",
    "df_parsed = df_stream.selectExpr(\"CAST(value AS STRING) as json_payload\") \\\n",
    "    .select(from_json(col(\"json_payload\"), booking_schema).alias(\"data\")) \\\n",
    "    .select(\"data.*\")\n",
    "    \n",
    "# Limpeza, LGPD e Cashback\n",
    "df_clean_cpf = df_parsed.withColumn(\"cpf_raw\", regexp_replace(col(\"primary_guest_cpf\"), r\"[.-]\", \"\"))\n",
    "df_transformed = df_clean_cpf.withColumn(\n",
    "    \"masked_cpf\",\n",
    "    expr(\"concat('***.', substring(cpf_raw, 4, 3), '.', substring(cpf_raw, 7, 3), '-**')\")\n",
    ").withColumn(\n",
    "    \"phone_clean\",\n",
    "    regexp_replace(col(\"primary_guest_phone\"), r\"[^\\d+]\", \"\")\n",
    ").withColumn(\n",
    "    \"earned_cashback\",\n",
    "    when(col(\"payment_type\") == \"pix\", col(\"total_price\") * 0.02)\n",
    "    .otherwise(col(\"total_price\") * 0.01)\n",
    ").select(\"timestamp\", \"id\", \"primary_guest_name\", \"masked_cpf\", \"phone_clean\", \"payment_type\", \"total_price\", \"earned_cashback\")\n",
    "\n",
    "# Gravar fluxo em memória para consultar com Spark SQL\n",
    "query = df_transformed.writeStream \\\n",
    "    .format(\"memory\") \\\n",
    "    .queryName(\"morada_stream_table\") \\\n",
    "    .outputMode(\"append\") \\\n",
    "    .start()\n",
    "\n",
    "print(\"Spark Streaming iniciado! Gravando em tempo real na tabela 'morada_stream_table'.\")"
   ]
