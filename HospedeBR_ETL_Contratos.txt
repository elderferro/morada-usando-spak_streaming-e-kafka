    "# HospedeBR — Clone do Airbnb para o mercado brasileiro\n",
    "### SPEC BASED DEVELOPMENT — Contratos de dados para um pipeline de streaming (Kafka + Spark Structured Streaming)\n",
    "\n",
    "Este notebook segue a mesma arquitetura do laboratório de streaming (Kafka + Spark, arquitetura\n",
    "medalhão Bronze/Silver/Gold, contratos de validação) usado no exemplo de ETL, mas modelando um\n",
    "**marketplace de hospedagem local, pensado para competir com o Airbnb no Brasil**.\n",
    "\n",
    "**Diferenciais de negócio incorporados como eventos/contratos** (para dar uma vantagem competitiva\n",
    "real frente ao Airbnb):\n",
    "\n",
    "- **PIX como pagamento nativo**, com **repasse instantâneo (D+0)** ao anfitrião — o Airbnb hoje repassa em D+1/D+7.\n",
    "- **Comissão zero** nos 3 primeiros meses para anfitriões migrando de outras plataformas (`HostMigrationBonusGranted`).\n",
    "- **Verificação de identidade via CPF + selfie** (mais confiança, menos fraude — dor conhecida do Airbnb no Brasil).\n",
    "- **Selo de Turismo de Base Comunitária** — hospedagens em comunidades/favelas com curadoria e repasse social (evento `CommunityTourismCertified`), algo que o Airbnb não trata como categoria de primeira classe.\n",
    "- **Selo de Sustentabilidade** (`SustainabilityBadgeAwarded`) com benefício de destaque na busca.\n",
    "- **Cashback em PIX** e **programa de fidelidade** (`CashbackIssued`, `LoyaltyPointsEarned`).\n",
    "- **Precificação sazonal para Carnaval, Réveillon e Festas Juninas**, calculada nos eventos de reserva.\n",
    "- **Suporte via WhatsApp** como canal oficial de atendimento (metadado no evento de suporte).\n",
    "- **Contratos de dados (data contracts)** validando cada evento antes de ele virar dado confiável (camada Silver),\n",
    "  com quarentena automática para eventos inválidos — a mesma filosofia do `ETLAmazon.ipynb`, adaptada para hospitalidade."

    "Cada tipo de evento é um **contrato**: um conjunto de campos obrigatórios e regras de negócio que\n",

    "producer = KafkaProducer(\n",
    "    bootstrap_servers=BOOTSTRAP_SERVERS,\n",
    "    value_serializer=lambda v: json.dumps(v).encode(\"utf-8\"),\n",

    "    guests_pool, hosts_pool, properties_pool, reservations_pool = [], [], [], []\n",
    "\n",
    "    inicio = time.time()\n",
    "    while stream_ativo.is_set() and (time.time() - inicio) < RUN_SECONDS:\n",
    "        evento, invalido = gerar_evento_aleatorio(guests_pool, hosts_pool, properties_pool, reservations_pool)\n",
    "\n",

    "\n",
    "- **Bronze**: evento bruto, como chegou.\n",
    "- **Silver**: apenas eventos que passaram em `is_valid`.\n",
    "- **Quarentena**: eventos que falharam em algum contrato, junto com o JSON original para investigação.\n",
    "- **Gold**: agregações de negócio (reservas, pagamentos, anfitriões, avaliações, fidelidade),\n",
    "  acumuladas em memória a cada micro-lote para exibição ao final do notebook."

    "raw_stream = (\n",

    "# ---------------------------- 12.1 STREAM VALIDADO ----------------------------\n",
    "validated_stream = (\n",
    "    parsed_stream\n",

    "\n",
    "def _concat(base, novo):\n",
    "    if novo is None or novo.empty:\n",
    "        return base\n",
    "    return novo.copy() if base is None else pd.concat([base, novo], ignore_index=True)\n"

    "# ---------------------------- 12.3 FUNÇÃO foreachBatch ----------------------------\n",
    "def processar_micro_lote(micro_df, batch_id):\n",
    "    global reservations_dataframe, payments_dataframe, hosts_dataframe\n",
    "    global reviews_dataframe, loyalty_dataframe, quality_dataframe, quarantine_dataframe\n",
    "\n",
    "    micro_df.persist()\n",
    "    total = micro_df.count()\n",
    "    if total == 0:\n",
    "        micro_df.unpersist()\n",
    "        return\n",
    "\n",
    "    # BRONZE — evento bruto\n",
    "    (\n",
    "        micro_df\n",
    "        .select(\"raw_json\", \"kafka_timestamp\")\n",
    "        .withColumn(\"_batch_id\", F.lit(batch_id))\n",
    "        .write.mode(\"append\").parquet(str(BRONZE_PATH))\n",
    "    )\n",
    "\n",
    "    validos = micro_df.filter(F.col(\"is_valid\"))\n",
    "    invalidos = micro_df.filter(~F.col(\"is_valid\"))\n",
    "\n",
    "    # SILVER — apenas eventos válidos, \"achatados\"\n",
    "    silver_df = validos.select(\n",
    "        F.col(\"event.event_id\").alias(\"event_id\"),\n",
    "        F.col(\"event.event_type\").alias(\"event_type\"),\n",
    "        F.col(\"event.occurred_at\").alias(\"occurred_at\"),\n",
    "        F.col(\"event.guest_id\").alias(\"guest_id\"),\n",
    "        F.col(\"event.host_id\").alias(\"host_id\"),\n",
    "        F.col(\"event.property_id\").alias(\"property_id\"),\n",
    "        F.col(\"event.reservation_id\").alias(\"reservation_id\"),\n",
    "        F.col(\"event.data\").alias(\"data\"),\n",
    "    )\n",
    "    silver_df.write.mode(\"append\").parquet(str(SILVER_PATH))\n",
    "\n",
    "    # QUARENTENA — eventos inválidos com motivo\n",
    "    (\n",
    "        invalidos\n",
    "        .select(\"validation_error\", \"raw_json\")\n",
    "        .withColumn(\"_batch_id\", F.lit(batch_id))\n",
    "        .write.mode(\"append\").parquet(str(QUARANTINE_PATH))\n",
    "    )\n",
    "\n",
    "    with _batch_lock:\n",
    "        # GOLD — RESERVAS (com sazonalidade)\n",
    "        reservas_pd = (\n",
    "            silver_df.filter(F.col(\"event_type\") == \"ReservationCreated\")\n",
    "            .select(\n",
    "                data_value(\"season_tag\").alias(\"season_tag\"),\n",
    "                data_value(\"total_amount\").cast(\"double\").alias(\"total_amount\"),\n",
    "            ).toPandas()\n",
    "        )\n",
    "        reservations_dataframe = _concat(reservations_dataframe, reservas_pd)\n",
    "\n",
    "        # GOLD — PAGAMENTOS (peso do PIX vs demais métodos)\n",
    "        pagamentos_pd = (\n",
    "            silver_df.filter(F.col(\"event_type\") == \"PaymentApproved\")\n",
    "            .select(\n",
    "                data_value(\"payment_method\").alias(\"payment_method\"),\n",
    "                data_value(\"amount\").cast(\"double\").alias(\"amount\"),\n",
    "            ).toPandas()\n",
    "        )\n",
    "        payments_dataframe = _concat(payments_dataframe, pagamentos_pd)\n",
    "\n",
    "        # GOLD — ANFITRIÕES (repasse instantâneo + migração)\n",
    "        hosts_pd = (\n",
    "            silver_df.filter(F.col(\"event_type\").isin(\"InstantPayoutRequested\", \"HostMigrationBonusGranted\"))\n",
    "            .select(\n",
    "                F.col(\"event_type\"),\n",
    "                data_value(\"payout_amount\").cast(\"double\").alias(\"payout_amount\"),\n",
    "                data_value(\"plataforma_origem\").alias(\"plataforma_origem\"),\n",
    "            ).toPandas()\n",
    "        )\n",
    "        hosts_dataframe = _concat(hosts_dataframe, hosts_pd)\n",
    "\n",
    "        # GOLD — AVALIAÇÕES\n",
    "        reviews_pd = (\n",
    "            silver_df.filter(F.col(\"event_type\") == \"ReviewCreated\")\n",
    "            .select(data_value(\"rating\").cast(\"int\").alias(\"rating\"))\n",
    "            .toPandas()\n",
    "        )\n",
    "        reviews_dataframe = _concat(reviews_dataframe, reviews_pd)\n",
    "\n",
    "        # GOLD — FIDELIDADE / CASHBACK\n",
    "        loyalty_pd = (\n",
    "            silver_df.filter(F.col(\"event_type\").isin(\"CashbackIssued\", \"LoyaltyPointsEarned\"))\n",
    "            .select(\n",
    "                F.col(\"event_type\"),\n",
    "                data_value(\"cashback_amount\").cast(\"double\").alias(\"cashback_amount\"),\n",
    "                data_value(\"points\").cast(\"int\").alias(\"points\"),\n",
    "            ).toPandas()\n",
    "        )\n",
    "        loyalty_dataframe = _concat(loyalty_dataframe, loyalty_pd)\n",
    "\n",
    "        # GOLD — QUALIDADE DOS DADOS\n",
    "        quality_pd = (\n",
    "            micro_df.groupBy(\"is_valid\", \"validation_error\")\n",
    "            .agg(F.count(\"*\").alias(\"events\"))\n",
    "            .toPandas()\n",
    "        )\n",
    "        quality_dataframe = _concat(quality_dataframe, quality_pd)\n",
    "\n",
    "        # QUARENTENA (amostra para exibição)\n",
    "        quarentena_pd = invalidos.select(\"validation_error\", \"raw_json\").limit(5).toPandas()\n",
    "        quarantine_dataframe = _concat(quarantine_dataframe, quarentena_pd)\n",
    "\n",
    "    print(f\"MICRO-LOTE {batch_id}: total={total}, válidos={validos.count()}, quarentena={invalidos.count()}\")\n",
    "    micro_df.unpersist()\n"
   ]
