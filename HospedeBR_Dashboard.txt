    "## 2. Transformações de LGPD e cashback (reaproveitadas do `morada_auto_stream.ipynb`)\n",
    "\n",
    "O notebook `morada_auto_stream.ipynb` fazia isso com `regexp_replace`/`expr` do Spark, dentro do\n",

    "# ---------------------------- 2. LGPD (MASCARAMENTO) E CASHBACK ----------------------------\n",
    "def mascarar_cpf(cpf):\n",
    "    digitos = re.sub(r\"\\D\", \"\", cpf)\n",
    "    return f\"***.{digitos[3:6]}.{digitos[6:9]}-**\"\n",
    "\n",
    "def limpar_telefone(telefone):\n",
    "    return re.sub(r\"[^\\d+]\", \"\", telefone)\n",
    "\n",
    "def calcular_cashback(total_price, payment_type):\n",
    "    # Mesma regra do morada_auto_stream: 2% no PIX, 1% nos demais métodos de pagamento.\n",
    "    taxa = 0.02 if payment_type == \"PIX\" else 0.01\n",
    "    return round(total_price * taxa, 2)\n",
    "\n",
    "print(\"Funções de LGPD e cashback prontas: mascarar_cpf, limpar_telefone, calcular_cashback\")\n"
   ]

    "# ---------------------------- 3. CONTRATO DE RESERVA ----------------------------\n",
    "def contrato_reserva(evento):\n",
    "    return (\n",
    "        evento.get(\"guest_name\") is not None\n",
    "        and evento.get(\"payment_type\") in ALLOWED_PAYMENT_METHODS\n",
    "        and evento.get(\"season_tag\") in ALLOWED_SEASON_TAGS\n",
    "        and isinstance(evento.get(\"total_price\"), (int, float)) and evento[\"total_price\"] > 0\n",
    "        and isinstance(evento.get(\"rating\"), int) and 1 <= evento[\"rating\"] <= 5\n",
    "    )\n",
    "\n",
    "print(\"Contrato de reserva definido.\")\n"
   ]

    "## 4. Gerador de reservas (nome do hóspede, valor pago e avaliação)\n",

    "Cerca de 10% das reservas nascem propositalmente fora do contrato, só para mostrar a quarentena."

    "# ---------------------------- 4. GERADOR DE RESERVAS ----------------------------\n",
    "def gerar_reserva():\n",
    "    invalido = random.random() < 0.10\n",
    "    season = random.choices(ALLOWED_SEASON_TAGS, weights=[10, 10, 10, 70])[0]\n",
    "    multiplicador = {\"ALTA_TEMPORADA_CARNAVAL\": 2.2, \"ALTA_TEMPORADA_REVEILLON\": 2.5,\n",
    "                      \"FESTA_JUNINA\": 1.4, \"TEMPORADA_REGULAR\": 1.0}[season]\n",
    "    noites = random.randint(1, 10)\n",
    "    diaria = round(random.uniform(90, 900), 2)\n",
    "\n",
    "    reserva = {\n",
    "        \"id\": str(random.randint(100000, 999999)),\n",
    "        \"guest_name\": random.choice(NOMES_HOSPEDES),\n",
    "        \"guest_cpf\": f\"{random.randint(100,999)}.{random.randint(100,999)}.{random.randint(100,999)}-{random.randint(10,99)}\",\n",
    "        \"guest_phone\": f\"+55 (11) 9{random.randint(8000,9999)}-{random.randint(1000,9999)}\",\n",
    "        \"season_tag\": season,\n",
    "        \"total_price\": round(diaria * noites * multiplicador, 2),\n",
    "        \"payment_type\": random.choices(ALLOWED_PAYMENT_METHODS, weights=[50, 22, 13, 5, 10])[0],\n",
    "        \"rating\": random.randint(1, 5),\n",
    "        \"timestamp\": (datetime.now() - timedelta(minutes=random.randint(0, 5000))).strftime(\"%Y-%m-%d %H:%M:%S\"),\n",
    "    }\n",
    "\n",
    "    if invalido:\n",
    "        campo = random.choice([\"total_price\", \"payment_type\", \"rating\"])\n",
    "        reserva[campo] = None\n",
    "\n",
    "    return reserva\n",
    "\n",
    "reservas_brutas = [gerar_reserva() for _ in range(N_RESERVAS)]\n",
    "print(f\"{len(reservas_brutas)} reservas geradas (BRONZE).\")\n"

    "# ---------------------------- 5. BRONZE / SILVER / QUARENTENA ----------------------------\n",
    "bronze = reservas_brutas\n",
    "quarentena = [r for r in reservas_brutas if not contrato_reserva(r)]\n",
    "\n",
    "silver = []\n",
    "for r in reservas_brutas:\n",
    "    if not contrato_reserva(r):\n",
    "        continue\n",
    "    r_silver = dict(r)\n",
    "    r_silver[\"masked_cpf\"] = mascarar_cpf(r[\"guest_cpf\"])\n",
    "    r_silver[\"phone_clean\"] = limpar_telefone(r[\"guest_phone\"])\n",
    "    r_silver[\"earned_cashback\"] = calcular_cashback(r[\"total_price\"], r[\"payment_type\"])\n",
    "    del r_silver[\"guest_cpf\"]  # dado sensível não sai da camada bronze\n",
    "    del r_silver[\"guest_phone\"]\n",
    "    silver.append(r_silver)\n",
    "\n",
    "print(f\"BRONZE (bruto):      {len(bronze)} reservas\")\n",
    "print(f\"SILVER (válidas):    {len(silver)} reservas\")\n",
    "print(f\"QUARENTENA (falhas): {len(quarentena)} reservas\")\n",
    "\n",
    "silver_df = pd.DataFrame(silver)\n",
    "print(\"\\nAmostra da tabela Silver (já com CPF/telefone mascarados e cashback calculado):\")\n",
    "print(silver_df[[\"guest_name\", \"masked_cpf\", \"phone_clean\", \"payment_type\", \"total_price\", \"earned_cashback\", \"rating\"]].head(5))\n"
